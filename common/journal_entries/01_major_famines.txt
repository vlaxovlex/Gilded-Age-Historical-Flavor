je_famine = {
	icon = "gfx/interface/icons/event_icons/event_skull.dds"

	group = je_group_crises

	is_shown_when_inactive = {
		game_date >= 1837	#HJE edit 1937
		NOT = { has_variable = completed_major_famine_var }
		NOT = { is_country_type = decentralized }
		any_scope_state = {
			average_sol <= 10
		}
	}

	possible = {
		institution_investment_level = {
			institution = institution_social_security
			value < 4
		}
		OR = {
			any_scope_state = {
				OR = {
				   custom_tooltip = {
				      text = natural_disaster_variable
					  OR = {
						has_variable = natural_disaster_variable
						has_modifier = potato_famine_state_modifier		#HJE edit
					  }
				   }
		           custom_tooltip = {
				      text = grain_more_than_35_percent_more_expensive_tt
				      sg:grain = {
		                 state_goods_pricier >= 0.7
					  }
		          }
		        }
		        OR = {
				   AND = {
				      is_incorporated = yes
						average_sol <= 10	#HJE edit start
						any_scope_building = {
							is_farm_building = yes
							occupancy >= 0.1
							weekly_profit <= 0.5
						}
		           }						#HJE edit end
		           AND = {
				      is_incorporated = no
		              average_sol <= 5
		           }
		        }
			}
		}
	}

	immediate = {
		if = {
			limit = {
				any_scope_state = {
					average_sol < 10
					is_incorporated = yes
					custom_tooltip = {
						text = natural_disaster_variable
						OR = {												#HJE edit
							has_variable = natural_disaster_variable
							has_modifier = potato_famine_state_modifier		#HJE edit
						}													#HJE edit
					}
				}
			}
			random_scope_state = {
				limit = {
					average_sol <= 10
					is_incorporated = yes
					custom_tooltip = {
						text = natural_disaster_variable
						OR = {												#HJE edit
							has_variable = natural_disaster_variable
							has_modifier = potato_famine_state_modifier		#HJE edit
						}													#HJE edit
					}
				}
				save_scope_as = famine_state
				set_variable = famine_var
			}
			trigger_event = { id = famine_events.1 }
		}
		else_if = {
			limit = {
				any_scope_state = {
					average_sol <= 5
					OR = {	#HJE edit start
						any_scope_building = {
							is_farm_building = yes
							occupancy >= 0.1
							weekly_profit <= 0.5
						}	#HJE edit end
						sg:grain = {
							state_goods_pricier > 0.35 
						}
					}		#HJE edit		
				}
			}
			random_scope_state = {
				limit = {
					average_sol <= 5
					OR = {	#HJE edit start
						any_scope_building = {
							is_farm_building = yes
							occupancy >= 0.1
							weekly_profit <= 0.5
						}	#HJE edit end
						sg:grain = {
							state_goods_pricier > 0.35 
						}
					}		#HJE edit 
				}
				save_scope_as = famine_state
				set_variable = famine_var
			}
			trigger_event = { id = famine_events.1 }
		}
	}

	current_value = {
		value =	root.average_sol
	}

	goal_add_value = 3

	complete = {
		scope:journal_entry = {
			custom_tooltip = {
				text = je_major_famines_success
				is_goal_complete = yes
			}
		}
	}

	on_complete = {
		set_variable = {
			name = completed_major_famine_var
			months = normal_modifier_time
		}
		every_scope_state = {
			limit = {
			  has_modifier = great_famine_modifier
			}
			remove_modifier = great_famine_modifier
		}
		every_scope_state = {
			limit = {
			  has_modifier = emergency_famine_modifier
			}
			remove_modifier = emergency_famine_modifier
		}
		remove_variable = neighbor_country_1_var
		remove_variable = famine_donations_var
	}

	on_monthly_pulse = {
		random_events = {
			100 = 0
			10 = famine_events.2
			10 = famine_events.3
			10 = famine_events.4
			10 = famine_political.1
			10 = famine_political.2
			10 = famine_political.3
			10 = famine_political.4
			10 = famine_political.5
		}
	}

	timeout = 1825

	weight = 200

	should_be_pinned_by_default = yes

	transferable = no

	progressbar = yes
}
