# Scandinavian Pan-Nationalism
scandinavian_pan_nationalism = {
	icon = "gfx/interface/icons/event_icons/event_military.dds"

	group = je_group_historical_content

	is_shown_when_inactive = {
		OR = {
			AND = {
				exists = c:SWE
				c:SWE = THIS
			}
			AND = {
				exists = c:DEN
				c:DEN = THIS
			}
			AND = {
				exists = c:NOR
				c:NOR = THIS
			}
			AND = {
				exists = c:FIN
				c:FIN = THIS
			}
		}
		NOT = { has_global_variable = nordic_pan_nationalism_var }
	}
	
	possible = {
		is_subject = no
		any_scope_building = {
			is_building_type = building_university
		}
	}
	
	complete = {
		has_technology_researched = nationalism
		any_scope_building = {
			is_building_type = building_university
			occupancy > 0.9
			level >= 3
		}
	}
	
	on_complete = {
		set_global_variable = nordic_pan_nationalism_var
		c:SWE = {
			trigger_event = { id = scandinavia_flavor.2 }
		}
		c:DEN = {
			trigger_event = { id = scandinavia_flavor.2 }
		}
		c:NOR = {
			trigger_event = { id = scandinavia_flavor.2 }
		}
		c:FIN = {
			trigger_event = { id = scandinavia_flavor.2 }
		}
	}
	
	weight = 100
}

	}
	
	on_complete = {
		if = {
			limit = { has_law = law_type:law_free_trade }

				}
			}

		}
	}
	
	weight = 250
