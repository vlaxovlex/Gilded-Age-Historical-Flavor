GER = {
	use_culture_states = yes
	
	is_major_formation = yes
	
	unification_play = dp_unify_germany
	leadership_play = dp_leadership_germany

	required_states_fraction = 0.74	#HJE edit 0.75
	
	ai_will_do = { has_technology_researched = nationalism }

	possible = {
		OR = {
			has_technology_researched = pan-nationalism
			custom_tooltip = {
				text = je_german_unification_idea_trigger_desc
				has_variable = je_german_unification_idea
			}		
		}		
		custom_tooltip = {
			text = no_confederation_of_the_rhine_tt
			NOT = {
				exists = c:RHN
			}
		}	
		OR = {																	#HJE edit start
			country_has_primary_culture = cu:north_german			
			country_has_primary_culture = cu:south_german			
		}															
		any_country = {
			OR = {
				country_has_primary_culture = cu:north_german
				country_has_primary_culture = cu:south_german
			}
			has_technology_researched = nationalism
		}
		OR = {
			hidden_trigger = {
				year >= 1865
				is_ai = yes
			}
			custom_tooltip = {
				text = any_german_is_challenging_confederation_tt
				has_global_variable = GER_ladies_and_gentlemen_the_main_event
			}
		}																		#HJE edit end
	}
}
