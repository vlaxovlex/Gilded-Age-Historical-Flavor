HJE_italy.2 = {
	type = country_event
	placement = scope:RSI_sicily_scope
	title = HJE_italy.2.t
	desc = HJE_italy.2.d
	flavor = HJE_italy.2.f

	duration = 3

	event_image = {
		video = "gfx/event_pictures/southamerica_aristocrats.bk2"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/europenorthamerica/rich_and_poor"

	icon = "gfx/interface/icons/event_icons/event_default.dds"

	trigger = {
		NOT = { has_global_variable = sic_revolution }
		any_scope_state = { turmoil > 0.1 }
		any_scope_state = { state_region = s:STATE_SICILY }
		NOT = { has_variable = ITA_current_favourite_to_unify }
		OR = {
			country_has_primary_culture = cu:north_italian
			country_has_primary_culture = cu:south_italian
		}
		OR = {
			has_law = law_type:law_autocracy
			has_law = law_type:law_oligarchy
			has_law = law_type:law_landed_voting
		}
		OR = {
			owns_entire_state_region = STATE_LOMBARDY
			owns_entire_state_region = STATE_PIEDMONT
			owns_entire_state_region = STATE_UMBRIA
			owns_entire_state_region = STATE_CAMPANIA
			owns_entire_state_region = STATE_CALABRIA
			owns_entire_state_region = STATE_APULIA
			owns_entire_state_region = STATE_ABRUZZO
			owns_entire_state_region = STATE_SAVOY
			owns_entire_state_region = STATE_EMILIA
			owns_entire_state_region = STATE_TUSCANY
			owns_entire_state_region = STATE_ROMAGNA
			owns_entire_state_region = STATE_LAZIO
			owns_entire_state_region = STATE_VENETIA
			owns_entire_state_region = STATE_SICILY
		}
	}
	
	immediate = {
		set_global_variable = sic_revolution
		set_variable = risorgimento_anti_unifying_stance
		save_scope_as = SIC_two_sicilies_scope
		add_modifier = {
			name = revolutionary_zeal
			months = 12
		}
		random_scope_state = {
			limit = { state_region = s:STATE_SICILY }
			save_scope_as = RSI_sicily_scope
		}
		create_country = {
			tag = RSI
			origin = this
			state = scope:RSI_sicily_scope
			on_created = {
				set_variable = risorgimento_unifying_stance
				set_variable = pan_italian_rebellion
				activate_law = law_type:law_parliamentary_republic
				activate_law = law_type:law_wealth_voting
				create_military_formation = {
					type = army
					hq_region = c:RSI.capital.region
		
					combat_unit = {
						type = unit_type:combat_unit_type_irregular_infantry
						state_region = c:RSI.capital.state_region
						count = 5
					}
					
					combat_unit = {
						type = unit_type:combat_unit_type_hussars
						state_region = c:RSI.capital.state_region
						count = 5
					}
				}
			}
		}
		c:RSI = {
			create_diplomatic_play = {
				name = sicilian_revolt_1848_name
                target_country = ROOT
                type = dp_annex_war
				add_war_goal = {
					holder = ROOT
					type = annex_country
					target_country = c:RSI
					primary_demand = yes
				}
			}
		}
		add_radicals = {
			value = 0.4
			culture = cu:north_italian
		}
		add_radicals = {
			value = 0.4
			culture = cu:south_italian
		}
		random_interest_group = {
			limit = {
				is_marginal = no
				leader = { NOT = { has_ideology = ideology:ideology_radical } }
				OR = {
					is_interest_group_type = ig_rural_folk
					is_interest_group_type = ig_intelligentsia
					is_interest_group_type = ig_trade_unions
					is_interest_group_type = ig_petty_bourgeoisie
					is_interest_group_type = ig_industrialists
				}
			}
	    	leader = { set_ideology = ideology:ideology_radical }
			add_modifier = {
				name = ig_government_betrayed
				months = short_modifier_time
			}
		}
		random_interest_group = {
			limit = {
				is_marginal = no
				leader = { NOT = { has_ideology = ideology:ideology_radical } }
				OR = {
					is_interest_group_type = ig_rural_folk
					is_interest_group_type = ig_intelligentsia
					is_interest_group_type = ig_trade_unions
					is_interest_group_type = ig_petty_bourgeoisie
					is_interest_group_type = ig_industrialists
				}
			}
	    	leader = { set_ideology = ideology:ideology_radical }
			add_modifier = {
				name = ig_government_betrayed
				months = short_modifier_time
			}
		}
		random_interest_group = {
			limit = {
				is_marginal = no
				leader = { NOT = { has_ideology = ideology:ideology_radical } }
				OR = {
					is_interest_group_type = ig_rural_folk
					is_interest_group_type = ig_intelligentsia
					is_interest_group_type = ig_trade_unions
					is_interest_group_type = ig_petty_bourgeoisie

			add_modifier = {
				name = ig_government_betrayed
				months = short_modifier_time
			}
		}
	}

	option = {
		name = HJE_italy.2.a
 		play_as = c:RSI

	}		
}

	type = country_event
	placement = root
	title = HJE_italy.4.t
	desc = HJE_italy.4.d
	flavor = HJE_italy.4.f

	duration = 3

	event_image = {
		video = "gfx/event_pictures/unspecific_ruler_speaking_to_people.bk2"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/europenorthamerica/political_extremism"

	icon = "gfx/interface/icons/event_icons/event_newspaper.dds"

	trigger = {
		OR = {
			NOT = { exists = c:SAR }
			AND = {
				exists = c:SAR
				c:SAR = ROOT
			}
		}
		any_scope_state = { turmoil > 0.1 }
		#has_law = law_type:law_monarchy
		has_journal_entry = HJ_italian_1848_revolutions
		NOT = { has_global_variable = SAR_constitution_event }
		NOT = { 
			any_country = { 
				is_subject = no
				has_variable = ITA_current_favourite_to_unify 
			} 
		}
		OR = {
			country_has_primary_culture = cu:north_italian
			country_has_primary_culture = cu:south_italian
		}
		OR = {
			owns_entire_state_region = STATE_LOMBARDY
			owns_entire_state_region = STATE_PIEDMONT
			owns_entire_state_region = STATE_UMBRIA
			owns_entire_state_region = STATE_CAMPANIA
			owns_entire_state_region = STATE_CALABRIA
			owns_entire_state_region = STATE_APULIA
			owns_entire_state_region = STATE_ABRUZZO
			owns_entire_state_region = STATE_SAVOY
			owns_entire_state_region = STATE_EMILIA
			owns_entire_state_region = STATE_TUSCANY
			owns_entire_state_region = STATE_ROMAGNA
			owns_entire_state_region = STATE_LAZIO
			owns_entire_state_region = STATE_VENETIA
			owns_entire_state_region = STATE_SICILY
		}
		OR = {
			has_law = law_type:law_autocracy
			has_law = law_type:law_oligarchy
			has_law = law_type:law_landed_voting
		}
		NOT = {
			any_country = {
				NOT = { THIS = ROOT }
				country_definition = root.country_definition
			}
		}
	}

	immediate = {
		set_global_variable = SAR_constitution_event
	}

	option = {
		name = HJE_italy.4.a
		ai_chance = { base = 100 }
		default_option = yes
		set_variable = ITA_current_favourite_to_unify
		set_strategy = ai_strategy_unify_italy
		add_radicals = { value = -0.4 }
		add_loyalists = { value = 0.2 }
		activate_law = law_type:law_wealth_voting
		set_variable = risorgimento_unifying_stance
		set_ruling_interest_groups = { ig_intelligentsia ig_armed_forces }
		set_strategy = ai_strategy_progressive_agenda
		every_interest_group = {
			limit = { has_ideology = ideology:ideology_republican }
			remove_ideology = ideology:ideology_republican
		}
		if = {
			limit = {
				any_scope_character = {
					is_ruler = yes
					age > 35
				}
			}
			ruler = { kill_character = yes }
		}
		# reactionaries kinda angry
		ig:ig_landowners = { 
			add_modifier = {
				name = modifier_revolution_defeated_light
				months = short_modifier_time
			}
		}
		ig:ig_armed_forces = { 
			add_modifier = {
				name = modifier_revolution_defeated_light
				months = short_modifier_time
			}
		}
		hidden_effect = {
			every_country = {
				limit = {
					NOT = { THIS = ROOT }
					NOT = { is_subject_of = ROOT }
					NOT = { has_variable = pan_italian_rebellion }
					country_is_in_europe = yes
					OR = {
						country_has_primary_culture = cu:north_italian
						country_has_primary_culture = cu:south_italian
					}
				}
				change_relations = {
					country = ROOT
					value = -20
			    }
				set_secret_goal = {
					country = ROOT
					secret_goal = none
				}
				ROOT = {
					set_secret_goal = {
						country = PREV
						secret_goal = conquer
					}
				}
			}
		}
	}
	option = {
		name = HJE_italy.4.b
		ai_chance = { base = 0 }
		add_radicals = { value = 0.2 }
		add_loyalists = { value = -0.1 }
		every_interest_group = {
			limit = {
				OR = {
					has_ideology = ideology:ideology_radical
					leader = { has_ideology = ideology:ideology_radical }
				}
			}
			add_modifier = {
				name = ig_government_betrayed
				months = long_modifier_time
			}
		}
	}
}
