				any_scope_state = { state_region = s:STATE_SANTIAGO }
				NOT = { any_scope_state = { state_region = s:STATE_ANTOFAGASTA } }
			}
			AND = {

				NOT = { any_scope_state = { state_region = s:STATE_TARAPACA } }
			}
		}
	}
	
	immediate = {
		set_variable = had_war_of_the_pacific_event

			limit = { 
				NOT = { c:CHL = THIS }

				any_scope_state = { state_region = s:STATE_ANTOFAGASTA }
			}
			save_scope_as = chl_antofagasta_owner
		}
		scope:chl_antofagasta_owner = {
			random_scope_state = {
				limit = { state_region = s:STATE_ANTOFAGASTA }

				name = wartime_zeal
				months = 20
			}
		}
		if = {
			limit = {
				c:CHL = { is_player = no }
				NOT = { any_scope_state = { state_region = s:STATE_ANTOFAGASTA } }
			}
			c:CHL = {
				create_diplomatic_play = {
	                target_country = scope:chl_antofagasta_owner
	                type = dp_humiliation
					war = no
					add_war_goal = {
						holder = c:CHL
						type = return_state
						target_state = scope:chl_antofagasta

					}
				}
			}
