#Dredd Scott Case Splitoff
acw_sectionalism_startup.2 = {
	type = country_event
	
	event_image = {
		video = "gfx/event_pictures/unspecific_politicians_arguing.bk2"
	}
	
	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/asia/politician_parliament_motion"

	icon = "gfx/interface/icons/event_icons/event_scales.dds"
	
	title = acw_sectionalism_startup.2.t
	desc = acw_sectionalism_startup.2.d
	flavor = acw_sectionalism_startup.2.f
	
	duration = 3
	
	trigger = {
		exists = c:USA
		NOT = { has_law = law_type:law_slavery_banned }
		NOT = { has_variable = dredd_scott_case }
		year > 1854
		any_scope_state = {
			is_slave_state = no
			is_incorporated = yes
			any_neighbouring_state = {
				owner = ROOT
				is_slave_state = yes
			}
		}
	}
	
	immediate = {
		set_variable = {
			name = dredd_scott_case
			value = yes
		}
		random_scope_state = {
			limit = {
				is_slave_state = no
				is_incorporated = yes
				any_neighbouring_state = {
					owner = ROOT
					is_slave_state = yes
				}
			}
			save_scope_as = free_state
		}
	}
	
	#Scott wins the case
	option = {
	name = acw_sectionalism_startup.2.a 
		trigger_event = {
			id = acw_sectionalism_startup.3
			days = 5
		}
	}
	
	#Scott loses the case
	option = {
	name = acw_sectionalism_startup.2.b
		trigger_event = {
			id = acw_sectionalism_startup.4
			days = 5
		}
	}
}

	}
	
	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	title = acw_buildup.1.t
	desc = acw_buildup.1.d
	flavor = acw_buildup.1.f
	
	duration = 3
	
	trigger = {
		exists = c:USA
		NOT = { has_variable = house_gag_rule }
		NOT = { has_law = law_type:law_slavery_banned }
		year < 1837
	}
	
	immediate = {
		set_variable = {
			name = house_gag_rule
			value = yes
		}
		random_interest_group = { #slaver group
			limit = {
				is_revolutionary = no
				is_insurrectionary = no
				is_marginal = no
				OR = {
					has_ideology = ideology:ideology_pro_slavery
					leader = {
						has_ideology = ideology:ideology_slaver
					}
				}
			}
			leader = {
				save_scope_as = gag_representative
			}
		}
	}
	
	option = { #Enact the gag rule
	name = acw_buildup.1.a 
		custom_tooltip = { 
		set_variable = {
			name = gag_rule_texas
			value = yes
			}
		}
		add_modifier = {
			name = usa_house_gag_rule
			months = 72
			is_decaying = yes
		}
		every_interest_group = { #slavers happy
			limit = {
				OR = {
					has_ideology = ideology:ideology_pro_slavery
					leader = { has_ideology = ideology:ideology_slaver }
				}
				NOT = { has_modifier = ig_slavers_happy }
			}
			add_modifier = {
				name = ig_slavers_happy
				months = short_modifier_time
			}
		}
		every_interest_group = { #abolitionists mad
			limit = {
				OR = {
					has_ideology = ideology:ideology_anti_slavery
					leader = { has_ideology = ideology:ideology_abolitionist }
				}
				NOT = { has_modifier = ig_abolitionists_mad }
			}
			add_modifier = {
				name = ig_abolitionists_mad
				months = short_modifier_time
			}
		}
	}
