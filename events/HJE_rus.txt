	}

	immediate = {
		set_global_variable = RUS_won_the_great_game
		set_mutual_secret_goal = {
			country = c:GBR
			secret_goal = none
		}
		change_relations = {
			country = c:GBR
			value = 20
		}
		c:GBR = {

				country = ROOT
				value = 20
			}
		}

		name = HJE_russia.2.a
		default_option = yes

			name = modifier_rus_great_game_won
			months = -1
		}
		if = {
			limit = { 
				exists = c:PER 
				c:PER = {
					OR = {
						is_subject = no

					}
				}
			}
			create_diplomatic_pact = {
				country = c:PER
				type = dominion
			}
		}
		if = {
			limit = { 
				exists = c:AFG 
				c:AFG = {
					OR = {
						is_subject = no
						is_subject_of = c:GBR
					}
				}
			}
			create_diplomatic_pact = {
				country = c:AFG
				type = dominion
			}
		}
		else_if = {
			limit = {
				any_country = {
					is_afghan_nation = yes
					OR = {
						is_subject = no
						is_subject_of = c:GBR
					}
				}
			}
			create_diplomatic_pact = {
				country = PREV
				type = dominion
			}
		}	
		if = {
			limit = { exists = c:GBR }
			c:GBR = {
				every_subject_or_below = {
					limit = {
						OR = {
							country_is_in_central_asia = yes
							country_is_in_india = yes
						}
					}
					make_independent = yes
				}
			}
		}
		every_state = {
			limit = { 
				NOT = { has_claim_by = ROOT }
				NOT = { owner = ROOT }
				OR = {
					state_is_in_central_asia = yes
					state_is_in_india = yes
				}
			}
			this.state_region = { add_claim = ROOT }
		}
	}
}
